import java.util.* ;
import java.io.*; 

public class Solution {
	public static String[] NumberPattern(int n) {
		String[]temp=new String[n];
		int num=1;
		for(int i=0;i<n;i++){
			String row="";
			for(int j=0;j<n;j++){
				row=row+num+" ";
				num++;
			}
			temp[i]=row;
		}
		String[]result=new String[n];
		int index=0;
		for(int i=0;i<n;i+=2){
			result[index++]=temp[i];
		}
		for(int i=n-1;i>=0;i--){
			if(i%2!=0){
				result[index++]=temp[i];
			}
		}
		return result;
	}
}
